Backend — Node.js + Socket.io Server
backend/server.js
const express = require('express');
const http = require('http');
const { Server } = require('socket.io');
const mongoose = require('mongoose');
require('dotenv').config();

const app = express();
const server = http.createServer(app);
const io = new Server(server, {
  cors: { origin: "*", methods: ["GET", "POST"] }
});

// MongoDB Message Schema
const MessageSchema = new mongoose.Schema({
  username: String,
  message: String,
  timestamp: { type: Date, default: Date.now }
});
const Message = mongoose.model('Message', MessageSchema);

// Connect to MongoDB Atlas (free M0)
mongoose.connect(process.env.MONGO_URI);

// Health check endpoint (Prometheus scrapes this)
app.get('/health', (req, res) => res.json({ status: 'ok', uptime: process.uptime() }));

// WebSocket events
io.on('connection', (socket) => {
  console.log('User connected:', socket.id);

  socket.on('sendMessage', async (data) => {
    const msg = await Message.create(data);
    io.emit('newMessage', msg); // broadcast to ALL users
  });

  socket.on('disconnect', () => console.log('User left'));
});

server.listen(process.env.PORT || 3001, () => {
  console.log(`🚀 DreamChat server on port 3001`);
});

frontend/src/App.js — DreamChat UI
import { useState, useEffect, useRef } from 'react';
import io from 'socket.io-client';

const BACKEND = process.env.REACT_APP_BACKEND_URL || 'http://localhost:3001';
const socket = io(BACKEND);

function App() {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState('');
  const [username] = useState(`User${Math.floor(Math.random()*1000)}`);
  const endRef = useRef(null);

  useEffect(() => {
    socket.on('newMessage', (msg) => {
      setMessages(prev => [...prev, msg]);
      endRef.current?.scrollIntoView({ behavior: 'smooth' });
    });
    return () => socket.off('newMessage');
  }, []);

  const sendMessage = () => {
    if (!input.trim()) return;
    socket.emit('sendMessage', { username, message: input });
    setInput('');
  };

  return (
    <div className="chat-app">
      <h1>💬 DreamChat</h1>
      <div className="messages">
        {messages.map((m, i) => (
          <div key={i} className={m.username === username ? 'me' : 'other'}>
            <strong>{m.username}</strong>: {m.message}
          </div>
        ))}
        <div ref={endRef} />
      </div>
      <input value={input} onChange={e => setInput(e.target.value)}
        onKeyPress={e => e.key === 'Enter' && sendMessage()} />
      <button onClick={sendMessage}>Send</button>
    </div>
  );
}
export default App;

🚨 Common Docker Errors + Fixes

⛔ ERROR
Got permission denied while trying to connect to Docker daemon socket at unix:///var/run/docker.sock
✅ FIX
Your user isn't in the docker group yet.
sudo usermod -aG docker $USER
Then log out completely and log back in. Or: newgrp docker to apply immediately without logout.
⛔ ERROR
ERROR [frontend 5/6] RUN npm run build → Failed: react-scripts: not found
✅ FIX
You used npm ci --only=production which skips devDependencies (where react-scripts lives).
In your frontend Dockerfile, use RUN npm ci (no --only=production) in the build stage. Only use --only=production in the RUNTIME stage copy.
⛔ ERROR
WebSocket connection failed: ws://localhost:3001/socket.io — net::ERR_CONNECTION_REFUSED
✅ FIX
Frontend is trying to connect to localhost but backend is in a different container.
Set env var: REACT_APP_BACKEND_URL=http://backend:3001 in docker-compose (use service name, not localhost). For production K8s, use the backend Service DNS name.

# DreamChat Deploy Script — used in CI/CD + manual deploys

echo "✅ Deploy complete! Version: $IMAGE_TAG"

name: DreamChat CI/CD

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

env:
  PROJECT_ID: ${{ secrets.GCP_PROJECT_ID }}
  IMAGE_TAG: ${{ github.sha }} # unique per commit — critical!

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: {node-version: '20'}
      - name: Install + Test backend
        run: cd backend && npm ci && npm test

  security-scan:
    needs: test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Run Trivy vulnerability scan
        uses: aquasecurity/trivy-action@master
        with:
          scan-type: fs
          severity: CRITICAL,HIGH
          exit-code: '1' # FAIL if critical vulns found

  build-push:
    needs: security-scan
    if: github.ref == 'refs/heads/main' # only deploy from main
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: google-github-actions/auth@v2
        with:
          credentials_json: ${{ secrets.GCP_SA_KEY }}
      - name: Build + Push to GCR
        run: |

🚨 Error Bible — Fix Anything Fast

Every error you'll hit on this project + exact fix. Bookmark this section.

Kubernetes Errors

⛔ K8s
OOMKilled — pod killed due to Out Of Memory
✅ FIX
Increase memory limit in deployment.yaml:
limits: {memory: "512Mi"}
Also check for memory leaks in app: kubectl top pods -n dream-chat
⛔ K8s
Pending — 0/2 nodes available: Insufficient CPU
✅ FIX
Lower resource requests: requests: {cpu: "50m"} OR scale nodes: gcloud container clusters resize dream-cluster --num-nodes 3
⛔ K8s
WebSocket connections drop every 30 seconds on K8s
✅ FIX
K8s default service timeout is 30s. Add to Service annotations:
service.beta.kubernetes.io/aws-load-balancer-connection-idle-timeout: "3600"
Also enable sticky sessions for WebSocket load balancing.

CI/CD Errors

⛔ GHA
Error: buildx failed with: ERROR [auth] docker.io: unauthorized
✅ FIX
Add Docker credentials as GitHub Secrets:
Settings → Secrets → DOCKER_USERNAME and DOCKER_PASSWORD
Then in workflow: docker login -u ${{ secrets.DOCKER_USERNAME }}
⛔ Git
! [rejected] main → main (fetch first) — push rejected
✅ FIX
Remote has commits you don't have. Pull first:
git pull --rebase origin main
Then push again. NEVER use git push --force on shared branches!
⛔ Terraform
Error: Error creating Cluster: googleapi: Error 409: Already exists
✅ FIX
Cluster already exists. Either:
1. Import it: terraform import google_container_cluster.dream us-central1/dream-cluster
2. Or destroy old one first: gcloud container clusters delete dream-cluster
💰
Cost Saver Script — Run every night before sleep!
gcloud container clusters resize dream-cluster --num-nodes 0 --region us-central1
Scales cluster to 0 nodes = no node VM costs. In morning: --num-nodes 2 to bring back up. Saves ~$1.50/day = ~$45/month on GCP!
